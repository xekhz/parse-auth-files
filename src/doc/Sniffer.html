def initialize(np=50,t=20)
  @collected_data = Hash.new  
  Debugger.set_level(1)
  @number_packtes_to_capture = np
  @timeout=t
  Debugger.debug("\nSetting timeout #{@timeout} and packet capture number: #{@number_packtes_to_capture}",1)
  
  @pkt_array = Array.new
end
